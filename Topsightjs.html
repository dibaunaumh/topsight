topsight.define("station", function(model) {
    topsight.construct({
        "wheel": {
            center_x: 50,
            center_y: 50,
            radius: 20,
            color: "black",
            fill_color: "white"
        },
        "label": {
            x: 50,
            y: 10,
            text: model.name
        },
        "paper": {
            x: 50,
            y: 80,
            placeholder: "Enter description",
        }
    });
});

topsight.define("metro_line", function(model) {
    var prev_station = null;
    for (var i in model.stations) {
        var station_id = model.stations[i];
        var station = topsight.select("station#" + station_id);
        topsight.construct(station);
        if (prev_station) {
            topsight.construct({
                "pipe": {
                    from_x: prev_station.x,
                    from_y: prev_station.y,
                    to_x: station.x,
                    to_y: station.y,
                    color: model.color,
                    width: 2
                }
            });
            prev_station = station;
        }
    });
});

topsight.construct({
    station: [
        "south_kensington": {
            name: "South Kensington",
            x: 10,
            y: 70
        }
    ],
    metro_line: [
        "blue_line": {
            color: "blue",
            stations: ["south_kensington", "", ""]
        },
    ]
});
